Solve for a.
\begin{align*}
    Y &= a X + b \\
    Var(Y) &= Var(a X + b) \\
    Var(Y) &= a^2 Var(X) \\
    1 &= a^2 Var(X) \\
    1 &= a^2 \sigma^2 \\
    1 &= \pm a \sigma \\ 
    a &= \pm \frac{1}{\sigma} 
\end{align*}

Solve for b.
\begin{align*}
    Y &= aX + b \\
    E[Y] &= E[aX + b] \\
    E[Y] &= a E[X] + b \\
    0 &= a E[X] + b \\
    0 &= a \mu + b \\
    b &= -a \mu \\
    b &= - (\pm \frac{1}{\sigma}) \mu \\
    b &= \pm \frac{\mu}{\sigma}
\end{align*}
